CREATE OR REPLACE VIEW desglose_colaboradores AS SELECT colaborador.ID, CONCAT_WS(' ', colaborador.nombres, colaborador.apellido_paterno, colaborador.apellido_materno) AS nombre_completo, 
colaborador.fecha_nacimiento, colaborador.numero_retardos, colaborador.numero_desbloqueos,
carrera.nombre AS nombre_carrera, modalidad_colaborador.nombre AS nombre_modalidad, horario.hora_inicial, horario.hora_final
FROM colaborador JOIN carrera ON colaborador.ID_carrera = carrera.ID JOIN modalidad_colaborador ON colaborador.ID_modalidad = modalidad_colaborador.ID
JOIN horario ON colaborador.ID_horario = horario.ID;

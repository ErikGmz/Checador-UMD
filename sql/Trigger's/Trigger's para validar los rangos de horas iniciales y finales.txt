DELIMITER $$
	CREATE TRIGGER validarHorasChequeos
		BEFORE INSERT
		ON chequeo FOR EACH ROW
	BEGIN
		IF NEW.hora_inicial > NEW.hora_final THEN
			SIGNAL SQLSTATE '45000'
			SET MESSAGE_TEXT = 'La hora inicial no puede ser mayor a la final.';
		END IF;
	END$$    
DELIMITER ;
